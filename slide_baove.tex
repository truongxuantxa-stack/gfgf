\begin{frame}{Meal Planner – Tính khẩu phần bằng phương pháp khử Gauss}
    \textbf{Bài toán:} Tính định lượng các nguyên liệu đã chọn nhằm đáp ứng mục tiêu Protein, Carb và Fat của bữa ăn.
    
    \vspace{0.3cm}
    \begin{columns}[T]
        \begin{column}{0.4\textwidth}
            Mô hình toán học:
            \[ A\mathbf{x} = \mathbf{b} \]
            \small Trong đó:
            \begin{itemize}
                \item $A$: ma trận thành phần Protein, Carb và Fat của nguyên liệu.
                \item $\mathbf{x}$: vector khối lượng của ba nguyên liệu cần xác định.
                \item $\mathbf{b}$: mục tiêu macro của bữa ăn.
            \end{itemize}
            \vspace{0.2cm}
            \scriptsize \textit{*Với bữa chính có thêm nguyên liệu nhóm rau/fiber, hệ thống cố định định lượng rau trước, điều chỉnh mục tiêu còn lại rồi giải hệ $3\times3$.}
        \end{column}
        
        \begin{column}{0.55\textwidth}
